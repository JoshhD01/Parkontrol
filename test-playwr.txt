// Crear una factura en el modulo de facturacion

// 1. Creación de factura exitoso sin incluir el cliente en facturación

const playwright = require('playwright');
(async () => {
  const browser = await playwright['chromium'].launch({
    // headless: false, slowMo: 100, // Uncomment to visualize test
  });
  const page = await browser.newPage();

  // Load "http://localhost:4200/facturacion"
  await page.goto('http://localhost:4200/facturacion');

  // Resize window to 1536 x 799
  await page.setViewportSize({ width: 1536, height: 799 });

  // Click on <a> "receipt Facturacion"
  await page.click('[href="/facturacion"]');

  // Click on <div> "Facturas"
  await page.click('#mat-tab-group-0-label-1');

  // Click on <span> .mat-mdc-button-touch-target
  await page.click('.mat-mdc-button-touch-target');

  // Scroll wheel by X:0, Y:-432
  await page.mouse.wheel(0, -432);

  // Fill "3" on <input> #mat-input-60
  await page.fill('#mat-input-60', '3');

  // Click on <span> .cdk-focused > .mat-mdc-button-touch-target
  await page.click('.cdk-focused > .mat-mdc-button-touch-target');

  await browser.close();
})();
